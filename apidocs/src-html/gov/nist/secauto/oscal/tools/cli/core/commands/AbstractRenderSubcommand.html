029import gov.nist.secauto.metaschema.cli.processor.CLIProcessor.CallingContext;
030import gov.nist.secauto.metaschema.cli.processor.ExitCode;
031import gov.nist.secauto.metaschema.cli.processor.ExitStatus;
032import gov.nist.secauto.metaschema.cli.processor.InvalidArgumentException;
033import gov.nist.secauto.metaschema.cli.processor.OptionUtils;
034import gov.nist.secauto.metaschema.cli.processor.command.AbstractTerminalCommand;
035import gov.nist.secauto.metaschema.cli.processor.command.DefaultExtraArgument;
036import gov.nist.secauto.metaschema.cli.processor.command.ExtraArgument;
037import gov.nist.secauto.metaschema.cli.processor.command.ICommandExecutor;
038import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;

040import org.apache.commons.cli.CommandLine;
041import org.apache.commons.cli.Option;
042import org.apache.logging.log4j.LogManager;
043import org.apache.logging.log4j.Logger;
044
045import java.io.File;
046import java.io.IOException;
047import java.nio.file.Files;
048import java.nio.file.Path;
049import java.nio.file.Paths;
050import java.util.Collection;
051import java.util.List;
052
053import javax.xml.transform.TransformerException;
054
055import edu.umd.cs.findbugs.annotations.NonNull;
056import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
057
058public abstract class AbstractRenderSubcommand
059    extends AbstractTerminalCommand {
060  private static final Logger LOGGER = LogManager.getLogger(AbstractRenderSubcommand.class);
061
062  @NonNull
063  private static final String COMMAND = "render";
064  @NonNull
065  private static final List<ExtraArgument> EXTRA_ARGUMENTS = ObjectUtils.notNull(List.of(
066      new DefaultExtraArgument("source file", true),
067      new DefaultExtraArgument("destination file", false)));
068
069  @NonNull
070  private static final Option OVERWRITE_OPTION = ObjectUtils.notNull(
071      Option.builder()
072          .longOpt("overwrite")
073          .desc("overwrite the destination if it exists")
074          .build());
075
076  @Override
077  public String getName() {
078    return COMMAND;
079  }
080
081  @SuppressWarnings("null")
082  @Override
083  public Collection<? extends Option> gatherOptions() {
084    return List.of(OVERWRITE_OPTION);
085  }
086
087  @Override
088  @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "unmodifiable collection and immutable item")
089  public List<ExtraArgument> getExtraArguments() {
090    return EXTRA_ARGUMENTS;
091  }
092
093  @Override
094  public void validateOptions(CallingContext callingContext, CommandLine cmdLine) throws InvalidArgumentException {
095    List<String> extraArgs = cmdLine.getArgList();
096    if (extraArgs.size() != 2) {
097      throw new InvalidArgumentException("Both a source and destination argument must be provided.");
098    }
099
100    File source = new File(extraArgs.get(0));
101    if (!source.exists()) {
102      throw new InvalidArgumentException("The provided source '" + source.getPath() + "' does not exist.");
103    }
104    if (!source.canRead()) {
105      throw new InvalidArgumentException("The provided source '" + source.getPath() + "' is not readable.");
106    }
107  }
108
109  @Override
110  public ICommandExecutor newExecutor(CallingContext callingContext, CommandLine cmdLine) {
111    return ICommandExecutor.using(callingContext, cmdLine, this::executeCommand);
112  }
113
114  @SuppressWarnings({
115      "PMD.OnlyOneReturn", // readability
116      "unused"
117  })
118  protected ExitStatus executeCommand(
119      @NonNull CallingContext callingContext,
120      @NonNull CommandLine cmdLine) {
121    List<String> extraArgs = cmdLine.getArgList();
122    Path destination = resolvePathAgainstCWD(ObjectUtils.notNull(Paths.get(extraArgs.get(1)))); // .toAbsolutePath();
123
124    if (Files.exists(destination)) {
125      if (!cmdLine.hasOption(OVERWRITE_OPTION)) {
126        return ExitCode.INVALID_ARGUMENTS.exitMessage(
127            String.format("The provided destination '%s' already exists and the '%s' option was not provided.",
128                destination,
129                OptionUtils.toArgument(OVERWRITE_OPTION)));
130      }
131      if (!Files.isWritable(destination)) {
132        return ExitCode.IO_ERROR.exitMessage("The provided destination '" + destination + "' is not writable.");
133      }
134    }
135
136    Path input = resolvePathAgainstCWD(ObjectUtils.notNull(Paths.get(extraArgs.get(0))));
137    try {
138      performRender(input, destination);
139    } catch (IOException | TransformerException ex) {
140      return ExitCode.PROCESSING_ERROR.exit().withThrowable(ex);
141    }
142
143    if (LOGGER.isInfoEnabled()) {
144      LOGGER.info("Generated HTML file: " + destination.toString());
145    }
146    return ExitCode.OK.exit();
147  }
148
149  protected abstract void performRender(Path input, Path result) throws IOException, TransformerException;
150}
