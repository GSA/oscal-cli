041import gov.nist.secauto.oscal.tools.cli.core.commands.mappingcollection.MappingCollectionCommand;

044import gov.nist.secauto.oscal.tools.cli.core.commands.profile.ProfileCommand;

046
047import java.util.List;
048
049import edu.umd.cs.findbugs.annotations.NonNull;
050
051@SuppressWarnings("PMD.ShortClassName")
052public final class CLI {
053  public static void main(String[] args) {
054    System.setProperty("java.util.logging.manager", "org.apache.logging.log4j.jul.LogManager");
055
056    ExitStatus status = runCli(args);
057    int exitCode = status.getExitCode().getStatusCode();
058    System.exit(exitCode);
059  }
060
061  @NonNull
062  public static ExitStatus runCli(String... args) {
063    List<IVersionInfo> versions = ObjectUtils.notNull(
064        List.of(
065            new OscalCliVersion(),
066            new LibOscalVersion(),
067            new OscalVersion(),
068            new MetaschemaJavaVersion(),
069            new MetaschemaVersion()));
070    CLIProcessor processor = new CLIProcessor("oscal-cli", versions);
071    processor.addCommandHandler(new CatalogCommand());
072    processor.addCommandHandler(new ProfileCommand());
073    processor.addCommandHandler(new ComponentDefinitionCommand());
074    processor.addCommandHandler(new SystemSecurityPlanCommand());
075    processor.addCommandHandler(new MappingCollectionCommand());
076    processor.addCommandHandler(new AssessmentPlanCommand());
077    processor.addCommandHandler(new AssessmentResultsCommand());
078    processor.addCommandHandler(new PlanOfActionsAndMilestonesCommand());
079    processor.addCommandHandler(new MetaschemaCommand());
080    return processor.process(args);
081  }
082
083  private CLI() {
084    // disable construction
085  }
086}
